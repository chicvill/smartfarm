### 2.4 ESP32 CO2 제어 코드

```cpp
// CO2 제어 기능 추가

// ============================================
// CO2 관련 핀 및 설정
// ============================================
#define CO2_SENSOR_RX 26    // MH-Z19B RX
#define CO2_SENSOR_TX 27    // MH-Z19B TX
#define CO2_VALVE_PIN 33    // 솔레노이드 밸브

// CO2 목표 농도 (ppm)
const int CO2_TARGET_MIN = 800;
const int CO2_TARGET_MAX = 1000;
const int CO2_HYSTERESIS = 100;  // 떨림 방지

// CO2 공급 제한
const int CO2_START_HOUR = 7;   // 07:00
const int CO2_END_HOUR = 17;    // 17:00

// CO2 센서 (MH-Z19B)
#include <MHZ19.h>
MHZ19 mhz19;
HardwareSerial co2Serial(2);

bool co2ValveState = false;
int currentCO2 = 0;

// ============================================
// Setup에 추가
// ============================================
void setup() {
  // ... 기존 코드 ...
  
  // CO2 센서 초기화
  co2Serial.begin(9600, SERIAL_8N1, CO2_SENSOR_RX, CO2_SENSOR_TX);
  mhz19.begin(co2Serial);
  mhz19.autoCalibration(false);  // 자동 보정 끔
  
  // CO2 밸브 핀
  pinMode(CO2_VALVE_PIN, OUTPUT);
  digitalWrite(CO2_VALVE_PIN, LOW);
  
  Serial.println("CO2 system initialized");
}

// ============================================
// CO2 센서 읽기
// ============================================
int readCO2() {
  int co2 = mhz19.getCO2();
  
  if (co2 < 0 || co2 > 5000) {
    Serial.println("CO2 sensor error");
    return -1;
  }
  
  return co2;
}

// ============================================
// CO2 자동 제어
// ============================================
void controlCO2() {
  // 1. 시간 체크
  int currentHour = getCurrentHour();
  if (currentHour < 0) return;
  
  bool withinOperatingHours = (currentHour >= CO2_START_HOUR && 
                               currentHour < CO2_END_HOUR);
  
  if (!withinOperatingHours) {
    if (co2ValveState) {
      digitalWrite(CO2_VALVE_PIN, LOW);
      co2ValveState = false;
      Serial.println("CO2 OFF (outside hours)");
      publishCO2Status("time_off");
    }
    return;
  }
  
  // 2. 환기 중이면 CO2 공급 중단
  // (환기팬 또는 창문 열림 상태 확인)
  if (isVentilating()) {
    if (co2ValveState) {
      digitalWrite(CO2_VALVE_PIN, LOW);
      co2ValveState = false;
      Serial.println("CO2 OFF (ventilating)");
      publishCO2Status("vent_off");
    }
    return;
  }
  
  // 3. CO2 농도 읽기
  currentCO2 = readCO2();
  if (currentCO2 < 0) return;
  
  Serial.print("Current CO2: ");
  Serial.print(currentCO2);
  Serial.println(" ppm");
  
  // 4. 히스테리시스 제어
  if (!co2ValveState && currentCO2 < CO2_TARGET_MIN) {
    // 낮으면 ON
    digitalWrite(CO2_VALVE_PIN, HIGH);
    co2ValveState = true;
    Serial.print("CO2 ON (");
    Serial.print(currentCO2);
    Serial.println(" ppm)");
    publishCO2Status("auto_on");
    
  } else if (co2ValveState && currentCO2 > CO2_TARGET_MAX) {
    // 높으면 OFF
    digitalWrite(CO2_VALVE_PIN, LOW);
    co2ValveState = false;
    Serial.print("CO2 OFF (");
    Serial.print(currentCO2);
    Serial.println(" ppm)");
    publishCO2Status("auto_off");
  }
}

// ============================================
// 환기 중인지 확인
// ============================================
bool isVentilating() {
  // 환기팬 릴레이 상태 확인
  // 또는 온도가 설정값보다 높은지 확인
  // 간단히 온도 기준으로 판단
  float temp = dht.readTemperature();
  if (isnan(temp)) return false;
  
  return (temp > 28.0);  // 28도 이상이면 환기 중으로 간주
}

// ============================================
// CO2 상태 발행
// ============================================
void publishCO2Status(const char* reason) {
  StaticJsonDocument<200> doc;
  doc["node"] = NODE_ID;
  doc["co2_state"] = co2ValveState;
  doc["co2_ppm"] = currentCO2;
  doc["reason"] = reason;
  doc["hour"] = getCurrentHour();
  
  char buffer[256];
  serializeJson(doc, buffer);
  
  String topic = "smartfarm/" + String(NODE_ID) + "/co2/status";
  mqtt.publish(topic.c_str(), buffer);
}

// ============================================
// Loop에 추가
// ============================================
unsigned long lastCO2Check = 0;
const long CO2_CHECK_INTERVAL = 30000;  // 30초

void loop() {
  // ... 기존 코드 ...
  
  // CO2 자동 제어
  if (currentMillis - lastCO2Check >= CO2_CHECK_INTERVAL) {
    lastCO2Check = currentMillis;
    controlCO2();
  }
}

// ============================================
// 센서 읽기에 CO2 추가
// ============================================
void readSensors() {
  // ... 기존 센서 ...
  
  int co2 = readCO2();
  
  StaticJsonDocument<500> doc;
  // ... 기존 데이터 ...
  
  JsonObject sensors = doc.createNestedObject("sensors");
  // ... 기존 센서 ...
  sensors["co2"] = co2;  // ★ CO2 추가
  
  JsonObject status = doc.createNestedObject("status");
  status["led_state"] = ledState;
  status["co2_state"] = co2ValveState;  // ★ CO2 상태
  
  // ... MQTT 발행 ...
}
```

```cpp
// 양액재배 모니터링 (EC/pH는 전용 컨트롤러가 제어)
// ESP32는 모니터링만 수행

// ============================================
// 양액 센서 핀
// ============================================
#define EC_SENSOR_PIN 36    // 아날로그
#define PH_SENSOR_PIN 39    // 아날로그
#define LEVEL_SENSOR_PIN 13 // 디지털 (플로트)

// ============================================
// 양액 데이터
// ============================================
float currentEC = 0;
float currentPH = 0;
bool tankLevelOK = true;

// ============================================
// EC 센서 읽기 (간이형)
// ============================================
float readEC() {
  int rawValue = analogRead(EC_SENSOR_PIN);
  // 전압 → EC 변환 (센서별로 다름)
  float voltage = rawValue * (3.3 / 4095.0);
  float ec = voltage * 2.0;  // 예시 변환식
  return ec;
}

// ============================================
// pH 센서 읽기 (간이형)
// ============================================
float readPH() {
  int rawValue = analogRead(PH_SENSOR_PIN);
  float voltage = rawValue * (3.3 / 4095.0);
  // pH = 7 @ 2.5V 기준
  float ph = 7.0 + ((2.5 - voltage) / 0.18);
  return ph;
}

// ============================================
// 양액 탱크 레벨 확인
// ============================================
bool checkTankLevel() {
  return digitalRead(LEVEL_SENSOR_PIN);
}

// ============================================
// 센서 읽기에 양액 데이터 추가
// ============================================
void readSensors() {
  // ... 기존 코드 ...
  
  currentEC = readEC();
  currentPH = readPH();
  tankLevelOK = checkTankLevel();
  
  StaticJsonDocument<600> doc;
  // ... 기존 센서 ...
  
  JsonObject hydro = doc.createNestedObject("hydroponics");
  hydro["ec"] = currentEC;
  hydro["ph"] = currentPH;
  hydro["tank_ok"] = tankLevelOK;
  
  // 경보 체크
  if (!tankLevelOK) {
    hydro["alert"] = "Low tank level!";
  }
  if (currentEC < 1.0 || currentEC > 1.8) {
    hydro["alert"] = "EC out of range!";
  }
  if (currentPH < 5.5 || currentPH > 6.5) {
    hydro["alert"] = "pH out of range!";
  }
  
  // ... MQTT 발행 ...
}
```

### 4.2 VPD (증기압포차) 기반 제어

```javascript
// VPD 계산 및 제어

function calculateVPD(temp, humidity) {
    // 포화증기압 (SVP) 계산
    const svp = 0.6108 * Math.exp((17.27 * temp) / (temp + 237.3));
    
    // 실제증기압 (AVP)
    const avp = svp * (humidity / 100);
    
    // VPD = SVP - AVP (kPa)
    const vpd = svp - avp;
    
    return vpd;
}

const temp = msg.payload.sensors.temperature;
const humidity = msg.payload.sensors.humidity;

const vpd = calculateVPD(temp, humidity);

// 상추 최적 VPD: 0.6-1.0 kPa
let action = {};

if (vpd < 0.6) {
    // VPD 너무 낮음 → 습도 낮춰야
    action = {
        ventilation: true,
        mist: false,
        message: "VPD too low - increase ventilation"
    };
    
} else if (vpd > 1.0) {
    // VPD 너무 높음 → 습도 높여야
    action = {
        ventilation: false,
        mist: true,
        message: "VPD too high - increase humidity"
    };
    
} else {
    // VPD 적정
    action = {
        ventilation: "auto",
        mist: "auto",
        message: "VPD optimal"
    };
}

msg.payload = {
    vpd: vpd,
    temp: temp,
    humidity: humidity,
    action: action
};

node.status({fill:"green", shape:"dot", text:`VPD: ${vpd.toFixed(2)} kPa`});

return msg;
```

### 4.3 AI 기반 병해 조기 감지

```javascript
// 센서 데이터 이상 패턴 감지

// 병해 징후 패턴:
// 1. 습도 급등 + 온도 변화 적음 → 곰팡이 위험
// 2. CO2 소비 급감 → 생육 둔화
// 3. EC 급변 → 양액 문제

const temp = msg.payload.sensors.temperature;
const humidity = msg.payload.sensors.humidity;
const co2 = msg.payload.sensors.co2;
const ec = msg.payload.hydroponics.ec;

// 이전 값 가져오기
const prev = flow.get('prev_sensors') || {
    temp: temp,
    humidity: humidity,
    co2: co2,
    ec: ec
};

// 변화율 계산
const tempChange = Math.abs(temp - prev.temp);
const humidityChange = Math.abs(humidity - prev.humidity);
const co2Change = Math.abs(co2 - prev.co2);
const ecChange = Math.abs(ec - prev.ec);

let alerts = [];

// 패턴 1: 곰팡이 위험
if (humidityChange > 10 && tempChange < 2 && humidity > 75) {
    alerts.push({
        type: "fungal_risk",
        severity: "medium",
        message: "High humidity without temp change - fungal risk",
        action: "Increase ventilation immediately"
    });
}

// 패턴 2: 생육 둔화
if (co2Change > 200 && co2 > prev.co2) {
    alerts.push({
        type: "slow_growth",
        severity: "low",
        message: "CO2 consumption decreased - growth slowdown",
        action: "Check plant health"
    });
}

// 패턴 3: 양액 문제
if (ecChange > 0.3) {
    alerts.push({
        type: "nutrient_issue",
        severity: "high",
        message: "Rapid EC change detected",
        action: "Check nutrient solution immediately"
    });
}

// 현재 값 저장
flow.set('prev_sensors', {temp, humidity, co2, ec});

if (alerts.length > 0) {
    msg.payload = {
        alerts: alerts,
        timestamp: new Date()
    };
    
    // 텔레그램 알림 전송
    node.send([null, msg]);  // 두 번째 출력으로
    
    node.status({fill:"red", shape:"dot", text:`${alerts.length} alerts`});
} else {
    node.status({fill:"green", shape:"dot", text:"Normal"});
}

return msg;
```
